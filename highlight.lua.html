Simple Syntax Highlighting

M.highlight (string)

Wrap corresponding parts of Lua-Code with HTML-Spans of the classes comment, string, escape, operator and keyword. Does not work with multiline comments and multiline strings, everything else should be fine. 

local highlight = require "highlight"

return highlight.highlight( ""
.. "function get(table, key)"
.. "    return table[key]"
.. "end"
)
"<span class=keyword>function</span><wbr>&nbsp;get<span class=operator>(</span>table,<wbr>&nbsp;key<span class=operator>)</span><wbr>&nbsp;<wbr>&nbsp;<wbr>&nbsp;<wbr>&nbsp;<span class=keyword>return</span><wbr>&nbsp;table<span class=operator>[</span>key<span class=operator>]</span><span class=keyword>end</span>"
